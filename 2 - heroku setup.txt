# crea un applicazione su heroku

heroku create



# aggiorna la produzione

git push heroku master
heroku run rake db:migrate

# avvia l'applicazione nel browser
# (potrebbe non funzionare al primo avvio, perchè non c'è una root route)

heroku open



# rinomina l'applicazione

heroku rename raspoappsupercool




# controllare eventuali problemi
 heroku logs


# operazioni da eseguire (spesso) per aggiornare la produzione

git push
git push heroku
heroku run rake db:migrate
